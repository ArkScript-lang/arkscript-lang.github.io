171  node.list().erase(node.constList().begin() + i);

173  // but `(macro)` would get entirely applied because it's in a list,
174  // thus we need to evaluate the node if we have list[i].list[0] as a macro
175  if (applied)
176  recurApply(node.list()[i]);

179  added_begin = true;
180  }
181

187  }
