125

130  auto file_last_write = static_cast<decltype(timestamp)>(
131  std::chrono::duration_cast<std::chrono::seconds>(

206  auto readNumber = [&, this] (std::size_t& i) -> uint16_t {
207  uint16_t x = (static_cast<uint16_t>(m_bytecode[i]) << 8); ++i;
208  uint16_t y = static_cast<uint16_t>(m_bytecode[i]);
209  return x + y;
210  };
211 
212  // read tables and check if bytecode is valid
213  if (!(m_bytecode.size() > 4 && m_bytecode[i++] == 'a' &&
214  m_bytecode[i++] == 'r' && m_bytecode[i++] == 'k' &&
215  m_bytecode[i++] == Instruction::NOP))
216  throwStateError("invalid format: couldn't find magic constant");
217 
218  uint16_t major = readNumber(i); i++;
219  uint16_t minor = readNumber(i); i++;
220  uint16_t patch = readNumber(i); i++;
221 
222  if (major != ARK_VERSION_MAJOR)
223  {
224  std::string str_version = Ark::Utils::toString(major) + "." +
225  Ark::Utils::toString(minor) + "." +
226  Ark::Utils::toString(patch);
227  std::string builtin_version = Ark::Utils::toString(ARK_VERSION_MAJOR) + "." +
228  Ark::Utils::toString(ARK_VERSION_MINOR) + "." +
229  Ark::Utils::toString(ARK_VERSION_PATCH);
230  throwStateError("Compiler and VM versions don't match: " + str_version + " and " + builtin_version);
231  }
232 
233  using timestamp_t = unsigned long long;
234  timestamp_t timestamp = 0;
235  auto aa = (static_cast<timestamp_t>(m_bytecode[ i]) << 56),
236  ba = (static_cast<timestamp_t>(m_bytecode[++i]) << 48),
237  ca = (static_cast<timestamp_t>(m_bytecode[++i]) << 40),
238  da = (static_cast<timestamp_t>(m_bytecode[++i]) << 32),
239  ea = (static_cast<timestamp_t>(m_bytecode[++i]) << 24),
240  fa = (static_cast<timestamp_t>(m_bytecode[++i]) << 16),
241  ga = (static_cast<timestamp_t>(m_bytecode[++i]) << 8),
242  ha = (static_cast<timestamp_t>(m_bytecode[++i]));
243  i++;
244  timestamp = aa + ba + ca + da + ea + fa + ga + ha;
245 
246  if (m_bytecode[i] == Instruction::SYM_TABLE_START)
247  {
248  i++;
249  uint16_t size = readNumber(i);
250  m_symbols.reserve(size);
251  i++;
252 
253  for (uint16_t j=0; j < size; ++j)
254  {
255  std::string symbol = "";
256  while (m_bytecode[i] != 0)
257  symbol.push_back(m_bytecode[i++]);
258  i++;
259 
260  m_symbols.push_back(symbol);
261  }
262  }
263  else
264  throwStateError("couldn't find symbols table");
265 
266  if (m_bytecode[i] == Instruction::VAL_TABLE_START)
267  {
268  i++;
269  uint16_t size = readNumber(i);
270  m_constants.reserve(size);
271  i++;
272

274  {
275  uint8_t type = m_bytecode[i];
276  i++;
277

279  {

282  val.push_back(m_bytecode[i++]);
283  i++;
284

286  }

291  val.push_back(m_bytecode[i++]);
292  i++;
293

295  }
296  else if (type == Instruction::FUNC_TYPE)
297  {

299  i++;
300  m_constants.emplace_back(addr);
301  i++; // skip NOP
302  }
303  else

305  }
306  }
307  else

309

311  {
312  i++;

314  m_plugins.reserve(size);
315  i++;
