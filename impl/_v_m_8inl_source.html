12  // reset ip and pp
13  m_ip = 0;
14  m_pp = 0;

16  // find id of function
17  auto it = std::find(m_state->m_symbols.begin(), m_state->m_symbols.end(), name);
18  if (it == m_state->m_symbols.end())
19  throwVMError("Couldn't find symbol with name " + name);
20 
21  // convert and push arguments in reverse order
22  std::vector<Value> fnargs { { args... } };
23  for (auto it2=fnargs.rbegin(), it_end=fnargs.rend(); it2 != it_end; ++it2)
24  m_frames.back().push(*it2);
25 
26  // find function object and push it if it's a pageaddr/closure
27  uint16_t id = static_cast<uint16_t>(std::distance(m_state->m_symbols.begin(), it));
28  auto var = findNearestVariable(id);
29  if (var != nullptr)
30  {
31  if (var->valueType() != ValueType::PageAddr && var->valueType() != ValueType::Closure)
32  throwVMError("Symbol " + name + " isn't a function");
33 
34  m_frames.back().push(*var);
35  m_last_sym_loaded = id;
36  }
37  else
38  throwVMError("Couldn't load symbol with name " + name);
39 
40  std::size_t frames_count = m_frames.size();
41  // call it
42  call(static_cast<int16_t>(sizeof...(Args)));
43  // reset instruction pointer, otherwise the safeRun method will start at ip = -1
44  // without doing m_ip++ as intended (done right after the call() in the loop, but here
45  // we start outside this loop)
46  m_ip = 0;
47

49  safeRun(/* untilFrameCount */ frames_count);
50

54  else

90 inline void VM::call(int16_t argc_)
91 {
92  /*
93  Argument: number of arguments when calling the function
94  Job: Call function from its symbol id located on top of the stack. Take the given number of
95  arguments from the top of stack and give them to the function (the first argument taken
96  from the stack will be the last one of the function). The stack of the function is now composed
97  of its arguments, from the first to the last one
98  */
99  using namespace Ark::internal;
100 
101  uint16_t argc = 0;
102 
103  // handling calls from C++ code
104  if (argc_ <= -1)
105  {
106  ++m_ip;
107  auto x = (static_cast<uint16_t>(m_state->m_pages[m_pp][m_ip]) << 8); ++m_ip;
108  auto y = static_cast<uint16_t>(m_state->m_pages[m_pp][m_ip]);
109  argc = x + y;
110  }
111  else
112  argc = argc_;
113 
114  Value function = *m_frames.back().pop();
115 
116  switch (function.valueType())
117  {
118  // is it a builtin function name?
119  case ValueType::CProc:
120  {
121  // drop arguments from the stack
122  std::vector<Value> args(argc);
123  for (uint16_t j=0; j < argc; ++j)
124  args[argc - 1 - j] = *m_frames.back().pop();
125 
126  // call proc
127  m_frames.back().push(function.proc()(args, this));
128  return;
129  }
130 
131  // is it a user defined function?
132  case ValueType::PageAddr:
133  {
134  int old_frame = static_cast<int>(m_frames.size()) - 1;
135  PageAddr_t new_page_pointer = function.pageAddr();
136 
137  // create dedicated frame
138  createNewScope();
139  m_frames.emplace_back(m_ip, static_cast<uint16_t>(m_pp), new_page_pointer);
140  // store "reference" to the function to speed the recursive functions
141  if (m_last_sym_loaded < m_state->m_symbols.size())
142  (*m_locals.back())[m_last_sym_loaded] = function;
143 
144  m_pp = new_page_pointer;
145  m_ip = -1; // because we are doing a m_ip++ right after that
146  for (std::size_t j=0; j < argc; ++j)
147  m_frames.back().push(*m_frames[old_frame].pop());
148  break;
149  }
150 
151  // is it a user defined closure?
152  case ValueType::Closure:
153  {
154  int old_frame = static_cast<int>(m_frames.size()) - 1;
155  Closure& c = function.closure_ref();
156  PageAddr_t new_page_pointer = c.pageAddr();
157 
158  // load saved scope
159  m_locals.push_back(c.scope());
160  // create dedicated frame
161  createNewScope();
162  m_frames.back().incScopeCountToDelete();
163  m_frames.emplace_back(m_ip, static_cast<uint16_t>(m_pp), new_page_pointer);
164 
165  m_pp = new_page_pointer;
166  m_ip = -1; // because we are doing a m_ip++ right after that
167  for (std::size_t j=0; j < argc; ++j)
168  m_frames.back().push(*m_frames[old_frame].pop());
169  break;
170  }
171 
172  default:
173  throwVMError("couldn't identify function object: type index " + Ark::Utils::toString(static_cast<int>(function.valueType())));
174  }
175 
176  // checking function arity
177  std::size_t received_argc = m_frames.back().stackSize();
178  std::size_t needed_argc = 0;
179  if (m_state->m_options & FeatureFunctionArityCheck)
180  {
181  std::size_t index = 0;
182  while (m_state->m_pages[m_pp][index] == Instruction::MUT)
183  {
184  needed_argc += 1;
185  index += 3; // jump the argument of MUT (integer on 2 bits, big endian)
186  }
187 
188  if (needed_argc != received_argc)
189  throwVMError("Function needs " + Ark::Utils::toString(needed_argc) + " arguments, and received " + Ark::Utils::toString(received_argc));
190  }
191 }
