29  void Compiler::compile()
30  {
31  /*
32  Generating headers:
33  - lang name (to be sure we are executing an ArkScript file)
34  on 4 bytes (ark + padding)
35  - version (major: 2 bytes, minor: 2 bytes, patch: 2 bytes)
36  - timestamp (8 bytes, unix format)
37  - symbols table header
38  + elements
39  - values table header
40  + elements
41  */
42 
43  if (m_debug >= 2)
44  Ark::logger.info("Adding magic constant");
45 
46  m_bytecode.push_back('a');
47  m_bytecode.push_back('r');
48  m_bytecode.push_back('k');
49  m_bytecode.push_back(Instruction::NOP);
50 
51  // push version
52  if (m_debug >= 2)
53  {
54  Ark::logger.info("Major: ", ARK_VERSION_MAJOR);
55  Ark::logger.info("Minor: ", ARK_VERSION_MINOR);
56  Ark::logger.info("Patch: ", ARK_VERSION_PATCH);
57  }
58  pushNumber(ARK_VERSION_MAJOR);
59  pushNumber(ARK_VERSION_MINOR);
60  pushNumber(ARK_VERSION_PATCH);
61 
62  // push timestamp
63  unsigned long long timestamp = std::chrono::duration_cast<std::chrono::seconds>(
64  std::chrono::system_clock::now().time_since_epoch()
65  ).count();
66  for (char c=0; c < 8; c++)
67  {
68  unsigned d = 56 - 8 * c;
69  uint8_t b = (timestamp & (0xff << d)) >> d;
70  m_bytecode.push_back(b);
71  if (m_debug >= 2)
72  std::cout << static_cast<int>(b) << " ";
73  }
74  if (m_debug >= 2)
75  std::cout << std::endl;
76 
77  if (m_debug >= 1)
78  Ark::logger.info("Timestamp: ", timestamp);
79 
80  if (m_debug >= 1)
81  Ark::logger.info("Adding symbols table header");
82 
83  // symbols table
84  m_bytecode.push_back(Instruction::SYM_TABLE_START);
85  if (m_debug >= 1)
86  Ark::logger.info("Compiling");
87  // gather symbols, values, and start to create code segments
88  m_code_pages.emplace_back(); // create empty page
89  _compile(m_parser.ast(), 0);
90  if (m_debug >= 1)
91  Ark::logger.info("Adding symbols table");
92  // push size
93  pushNumber(static_cast<uint16_t>(m_symbols.size()));
94  // push elements
95  for (auto sym : m_symbols)
96  {
97  // push the string, null terminated
98  for (std::size_t i=0, size=sym.size(); i < size; ++i)
99  m_bytecode.push_back(sym[i]);
100  m_bytecode.push_back(Instruction::NOP);
101  }
102 
103  if (m_debug >= 1)
104  Ark::logger.info("Adding constants table");
105 
106  // values table
107  m_bytecode.push_back(Instruction::VAL_TABLE_START);
108  // push size
109  pushNumber(static_cast<uint16_t>(m_values.size()));
110  // push elements (separated with 0x00)
111  for (auto val : m_values)
112  {
113  if (val.type == CValueType::Number)
114  {
115  m_bytecode.push_back(Instruction::NUMBER_TYPE);
116  auto n = std::get<double>(val.value);
117  std::string t = Ark::Utils::toString(n);
118  for (std::size_t i=0, size=t.size(); i < size; ++i)
119  m_bytecode.push_back(t[i]);
120  }
121  else if (val.type == CValueType::String)
122  {
123  m_bytecode.push_back(Instruction::STRING_TYPE);
124  std::string t = std::get<std::string>(val.value);
125  for (std::size_t i=0, size=t.size(); i < size; ++i)
126  m_bytecode.push_back(t[i]);
127  }
128  else if (val.type == CValueType::PageAddr)
129  {
130  m_bytecode.push_back(Instruction::FUNC_TYPE);
131  pushNumber(static_cast<uint16_t>(std::get<std::size_t>(val.value)));
132  }
133  else
134  {
135  throw std::runtime_error("CompilerError: trying to put a value in the value table, but the type isn't handled.\nCertainly a logic problem in the compiler source code");
136  }
137 
138  m_bytecode.push_back(Instruction::NOP);
139  }
140 
141  if (m_debug >= 1)
142  Ark::logger.info("Adding plugins table");
143 
144  // plugins table
145  m_bytecode.push_back(Instruction::PLUGIN_TABLE_START);
146  // push size
147  pushNumber(static_cast<uint16_t>(m_plugins.size()));
148  // push elements
149  for (auto plugin: m_plugins)
150  {
151  // push the string, null terminated
152  for (std::size_t i=0, size=plugin.size(); i < size; ++i)
153  m_bytecode.push_back(plugin[i]);
154  m_bytecode.push_back(Instruction::NOP);
155  }
156 
157  if (m_debug >= 1)
158  Ark::logger.info("Adding code segments");
159 
160  // start code segments
161  for (auto page : m_code_pages)
162  {
163  if (m_debug >= 2)
164  Ark::logger.info("-", page.size() + 1);
165 
166  m_bytecode.push_back(Instruction::CODE_SEGMENT_START);
167  // push number of elements
168  if (!page.size())
169  {
170  pushNumber(0x01);
171  m_bytecode.push_back(Instruction::HALT);
172  return;
173  }
174  pushNumber(static_cast<uint16_t>(page.size() + 1));
175 
176  for (auto inst : page)
177  m_bytecode.push_back(inst.inst);
178  // just in case we got too far, always add a HALT to be sure the
179  // VM won't do anything crazy
180  m_bytecode.push_back(Instruction::HALT);
181  }
182 
183  if (!m_code_pages.size())
184  {
185  m_bytecode.push_back(Instruction::CODE_SEGMENT_START);
186  pushNumber(static_cast<uint16_t>(1));
187  m_bytecode.push_back(Instruction::HALT);
188  }
189  }

206  void Compiler::_compile(const Ark::internal::Node& x, int p)
207  {
208  if (m_debug >= 2)
209  Ark::logger.info(x);
210 
211  // register symbols
212  if (x.nodeType() == Ark::internal::NodeType::Symbol)
213  {
214  std::string name = x.string();
215 
216  // check if 'name' isn't a builtin/operator name before pushing it as a 'var-use'
217  if (auto it_builtin = isBuiltin(name))
218  {
219  page(p).emplace_back(Instruction::BUILTIN);
220  pushNumber(static_cast<uint16_t>(it_builtin.value()), &page(p));
221  }
222  else if (auto it_operator = isOperator(name))
223  {
224  page(p).emplace_back(static_cast<Instruction>(Instruction::FIRST_OPERATOR + it_operator.value()));
225  }
226  else
227  {
228  std::size_t i = addSymbol(name);
229 
230  page(p).emplace_back(Instruction::LOAD_SYMBOL);
231  pushNumber(static_cast<uint16_t>(i), &page(p));
232  }
233 
234  return;
235  }
236  if (x.nodeType() == Ark::internal::NodeType::GetField)
237  {
238  std::string name = x.string();
239  // 'name' shouldn't be a builtin/operator, we can use it as-is
240  std::size_t i = addSymbol(name);
241 
242  page(p).emplace_back(Instruction::GET_FIELD);
243  pushNumber(static_cast<uint16_t>(i), &page(p));
244 
245  return;
246  }
247  // register values
248  if (x.nodeType() == Ark::internal::NodeType::String || x.nodeType() == Ark::internal::NodeType::Number)
249  {
250  std::size_t i = addValue(x);
251 
252  page(p).emplace_back(Instruction::LOAD_CONST);
253  pushNumber(static_cast<uint16_t>(i), &page(p));
254 
255  return;
256  }
257  // empty code block should be nil
258  if (x.const_list().empty())
259  {
260  auto it_builtin = isBuiltin("nil");
261  page(p).emplace_back(Instruction::BUILTIN);
262  pushNumber(static_cast<uint16_t>(it_builtin.value()), &page(p));
263  return;
264  }
265  // registering structures
266  if (x.const_list()[0].nodeType() == Ark::internal::NodeType::Keyword)
267  {
268  Ark::internal::Keyword n = x.const_list()[0].keyword();
269 
270  if (n == Ark::internal::Keyword::If)
271  {
272  // compile condition
273  _compile(x.const_list()[1], p);
274  // jump only if needed to the x.list()[2] part
275  page(p).emplace_back(Instruction::POP_JUMP_IF_TRUE);
276  std::size_t jump_to_if_pos = page(p).size();
277  // absolute address to jump to if condition is true
278  pushNumber(static_cast<uint16_t>(0x00), &page(p));
279  // else code
280  _compile(x.const_list()[3], p);
281  // when else is finished, jump to end
282  page(p).emplace_back(Instruction::JUMP);
283  std::size_t jump_to_end_pos = page(p).size();
284  pushNumber(static_cast<uint16_t>(0x00), &page(p));
285  // set jump to if pos
286  page(p)[jump_to_if_pos] = (static_cast<uint16_t>(page(p).size()) & 0xff00) >> 8;
287  page(p)[jump_to_if_pos + 1] = static_cast<uint16_t>(page(p).size()) & 0x00ff;
288  // if code
289  _compile(x.const_list()[2], p);
290  // set jump to end pos
291  page(p)[jump_to_end_pos] = (static_cast<uint16_t>(page(p).size()) & 0xff00) >> 8;
292  page(p)[jump_to_end_pos + 1] = static_cast<uint16_t>(page(p).size()) & 0x00ff;
293  }
294  else if (n == Ark::internal::Keyword::Set)
295  {
296  std::string name = x.const_list()[1].string();
297  std::size_t i = addSymbol(name);
298 
299  // put value before symbol id
300  _compile(x.const_list()[2], p);
301 
302  page(p).emplace_back(Instruction::STORE);
303  pushNumber(static_cast<uint16_t>(i), &page(p));
304  }
305  else if (n == Ark::internal::Keyword::Let)
306  {
307  std::string name = x.const_list()[1].string();
308  std::size_t i = addSymbol(name);
309 
310  // put value before symbol id
311  _compile(x.const_list()[2], p);
312 
313  page(p).emplace_back(Instruction::LET);
314  pushNumber(static_cast<uint16_t>(i), &page(p));
315  }
316  else if (n == Ark::internal::Keyword::Mut)
317  {
318  std::string name = x.const_list()[1].string();
319  std::size_t i = addSymbol(name);
320 
321  // put value before symbol id
322  _compile(x.const_list()[2], p);
323 
324  page(p).emplace_back(Instruction::MUT);
325  pushNumber(static_cast<uint16_t>(i), &page(p));
326  }
327  else if (n == Ark::internal::Keyword::Fun)
328  {
329  // capture, if needed
330  for (auto it=x.const_list()[1].const_list().begin(), it_end=x.const_list()[1].const_list().end(); it != it_end; ++it)
331  {
332  if (it->nodeType() == NodeType::Capture)
333  {
334  page(p).emplace_back(Instruction::CAPTURE);
335  std::size_t var_id = addSymbol(it->string());
336  pushNumber(static_cast<uint16_t>(var_id), &(page(p)));
337  }
338  }
339  // create new page for function body
340  m_code_pages.emplace_back();
341  std::size_t page_id = m_code_pages.size() - 1;
342  // load value on the stack
343  page(p).emplace_back(Instruction::LOAD_CONST);
344  std::size_t id = addValue(page_id); // save page_id into the constants table as PageAddr
345  pushNumber(static_cast<uint16_t>(id), &page(p));
346  // pushing arguments from the stack into variables in the new scope
347  for (auto it=x.const_list()[1].const_list().begin(), it_end=x.const_list()[1].const_list().end(); it != it_end; ++it)
348  {
349  if (it->nodeType() == NodeType::Symbol)
350  {
351  page(page_id).emplace_back(Instruction::MUT);
352  std::size_t var_id = addSymbol(it->string());
353  pushNumber(static_cast<uint16_t>(var_id), &(page(page_id)));
354  }
355  }
356  // push body of the function
357  _compile(x.const_list()[2], page_id);
358  // return last value on the stack
359  page(page_id).emplace_back(Instruction::RET);
360  }
361  else if (n == Ark::internal::Keyword::Begin)
362  {
363  for (std::size_t i=1, size=x.const_list().size(); i < size; ++i)
364  _compile(x.const_list()[i], p);
365  }
366  else if (n == Ark::internal::Keyword::While)
367  {
368  // save current position to jump there at the end of the loop
369  std::size_t current = page(p).size();
370  // push condition
371  _compile(x.const_list()[1], p);
372  // absolute jump to end of block if condition is false
373  page(p).emplace_back(Instruction::POP_JUMP_IF_FALSE);
374  std::size_t jump_to_end_pos = page(p).size();
375  // absolute address to jump to if condition is false
376  pushNumber(static_cast<uint16_t>(0x00), &page(p));
377  // push code to page
378  _compile(x.const_list()[2], p);
379  // loop, jump to the condition
380  page(p).emplace_back(Instruction::JUMP);
381  // abosolute address
382  pushNumber(static_cast<uint16_t>(current), &page(p));
383  // set jump to end pos
384  page(p)[jump_to_end_pos] = (static_cast<uint16_t>(page(p).size()) & 0xff00) >> 8;
385  page(p)[jump_to_end_pos + 1] = static_cast<uint16_t>(page(p).size()) & 0x00ff;
386  }
387  else if (n == Ark::internal::Keyword::Import)
388  {
389  for (auto it=x.const_list().begin() + 1, it_end=x.const_list().end(); it != it_end; ++it)
390  {
391  // load const, push it to the plugins table
392  addPlugin(*it);
393  }
394  }
395  else if (n == Ark::internal::Keyword::Quote)
396  {
397  // create new page for quoted code
398  m_code_pages.emplace_back();
399  std::size_t page_id = m_code_pages.size() - 1;
400  _compile(x.const_list()[1], page_id);
401  page(page_id).emplace_back(Instruction::RET); // return to the last frame
402 
403  // call it
404  std::size_t id = addValue(page_id); // save page_id into the constants table as PageAddr
405  // page(p).emplace_back(Instruction::SAVE_ENV);
406  page(p).emplace_back(Instruction::LOAD_CONST);
407  pushNumber(static_cast<uint16_t>(id), &page(p));
408  }
409  else if (n == Ark::internal::Keyword::Del)
410  {
411  // get id of symbol to delete
412  std::string name = x.const_list()[1].string();
413  std::size_t i = addSymbol(name);
414 
415  page(p).emplace_back(Instruction::DEL);
416  pushNumber(static_cast<uint16_t>(i), &page(p));
417  }
418 
419  return;
420  }
421 
422  // if we are here, we should have a function name
423  // push arguments first, then function name, then call it
424  m_temp_pages.emplace_back();
425  int proc_page = -static_cast<int>(m_temp_pages.size());
426  _compile(x.const_list()[0], proc_page); // storing proc
427  // trying to handle chained closure.field.field.field...
428  std::size_t n = 1;
429  while (n < x.const_list().size())
430  {
431  if (x.const_list()[n].nodeType() == Ark::internal::NodeType::GetField)
432  {
433  _compile(x.const_list()[n], proc_page);
434  n++;
435  }
436  else
437  break;
438  }
439  std::size_t proc_page_len = m_temp_pages.back().size();
440  // we know that operators take only 1 instruction, so if there are more
441  // it's a builtin/function
442  if (proc_page_len > 1)
443  {
444  // push arguments on current page
445  for (auto exp=x.const_list().begin() + n, exp_end=x.const_list().end(); exp != exp_end; ++exp)
446  _compile(*exp, p);
447  // push proc from temp page
448  for (auto&& inst : m_temp_pages.back())
449  page(p).push_back(inst);
450  m_temp_pages.pop_back();
451 
452  // call the procedure
453  page(p).push_back(Instruction::CALL);
454  // number of arguments
455  std::size_t args_count = 0;
456  for (auto it=x.const_list().begin() + 1, it_end=x.const_list().end(); it != it_end; ++it)
457  {
458  if (it->nodeType() != Ark::internal::NodeType::GetField &&
459  it->nodeType() != Ark::internal::NodeType::Capture)
460  args_count++;
461  }
462  pushNumber(static_cast<uint16_t>(args_count), &page(p));
463  }
464  else // operator
465  {
466  // retrieve operator
467  auto op_inst = m_temp_pages.back()[0];
468  m_temp_pages.pop_back();
469 
470  // push arguments on current page
471  std::size_t exp_count = 0;
472  for (std::size_t index=n, size=x.const_list().size(); index < size; ++index)
473  {
474  _compile(x.const_list()[index], p);
475 
476  if ((index + 1 < size &&
477  x.const_list()[index + 1].nodeType() != Ark::internal::NodeType::GetField &&
478  x.const_list()[index + 1].nodeType() != Ark::internal::NodeType::Capture) ||
479  index + 1 == size)
480  exp_count++;
481 
482  // in order to be able to handle things like (op A B C D...)
483  // which should be transformed into A B op C op D op...
484  if (exp_count >= 2)
485  page(p).push_back(op_inst);
486  }
487 
488  if (exp_count == 1)
489  page(p).push_back(op_inst);
490 
491  // need to check we didn't push the (op A B C D...) things for operators not supporting it
492  if (exp_count > 2)
493  {
494  switch (op_inst.inst)
495  {
496  // authorized instructions
497  case Instruction::ADD:
498  case Instruction::SUB:
499  case Instruction::DIV:
500  case Instruction::MUL:
501  case Instruction::MOD:
502  case Instruction::AND_:
503  case Instruction::OR_:
504  break;
505 
506  default:
507  throw std::runtime_error("CompilerError: can not create a chained expression (of length " + Utils::toString(exp_count) +
508  ") for operator `" + FFI::operators[static_cast<std::size_t>(op_inst.inst - Instruction::FIRST_OPERATOR)] + "' " +
509  "at node `" + Utils::toString(x) + "'");
510  }
511  }
512  }
513 
514  return;
515  }
