65  void Parser::sugar(std::vector<Token>& tokens)
66  {
67  std::size_t i = 0;
68  while (true)
69  {
70  std::size_t line = tokens[i].line;
71  std::size_t col = tokens[i].col;
72 
73  if (tokens[i].token == "{")
74  {
75  tokens[i] = Token(TokenType::Grouping, "(", line, col);
76  tokens.insert(tokens.begin() + i + 1, Token(TokenType::Keyword, "begin", line, col));
77  }
78  else if (tokens[i].token == "}" || tokens[i].token == "]")
79  tokens[i] = Token(TokenType::Grouping, ")", line, col);
80  else if (tokens[i].token == "[")
81  {
82  tokens[i] = Token(TokenType::Grouping, "(", line, col);
83  tokens.insert(tokens.begin() + i + 1, Token(TokenType::Identifier, "list", line, col));
84  }
85 
86  ++i;
87 
88  if (i == tokens.size())
89  break;
90  }
91  }

94  Node Parser::parse(std::list<Token>& tokens, bool authorize_capture, bool authorize_field_read)
95  {
96  using namespace std::string_literals;
97 
98  Token token = nextToken(tokens);
99 
100  // parse block
101  if (token.token == "(")
102  {
103  bool previous_token_was_lparen = true;
104 
105  // create a list node to host the block
106  Node block(NodeType::List);
107  block.setPos(token.line, token.col);
108 
109  // handle sub-blocks
110  if (tokens.front().token == "(")
111  {
112  block.push_back(parse(tokens));
113  previous_token_was_lparen = false;
114  }
115 
116  // take next token, we don't want to play with a "("
117  token = nextToken(tokens);
118 
119  // return an empty block
120  if (token.token == ")")
121  return block;
122 
123  // loop until we reach the end of the block
124  do
125  {
126  auto atomized = atom(token);
127  std::pair<std::size_t, std::size_t> warn_info = std::make_pair(
128  token.line, token.col
129  );
130 
131  if (std::find(m_warns.begin(), m_warns.end(), warn_info) == m_warns.end() &&
132  (atomized.nodeType() == NodeType::String || atomized.nodeType() == NodeType::Number ||
133  atomized.nodeType() == NodeType::List) &&
134  previous_token_was_lparen)
135  {
136  if ((m_options & FeatureDisallowInvalidTokenAfterParen) == 0)
137  {
138  m_warns.push_back(std::move(warn_info));
139  Ark::logger.warn("Found a possible ill-formed code line: invalid token after `(' (token: {0}, at {1}:{2}, in {3})"s, token.token, token.line, token.col, m_file);
140  }
141  else
142  throwParseError("Ill-formed code line: invalid token after `('", token);
143  }
144 
145  block.push_back(atomized);
146 
147  except(!tokens.empty(), "Invalid syntax: no more token to consume", m_last_token);
148  m_last_token = tokens.front();
149 
150  if (token.type == TokenType::Keyword)
151  {
152  if (token.token == "if")
153  {
154  auto temp = tokens.front();
155  // parse condition
156  if (temp.type == TokenType::Grouping)
157  block.push_back(parse(tokens));
158  else if (temp.type == TokenType::Identifier || temp.type == TokenType::Number ||
159  temp.type == TokenType::String)
160  block.push_back(atom(nextToken(tokens)));
161  else
162  throwParseError("ill-formed if, couldn't find condition", temp);
163  // parse 'then'
164  block.push_back(parse(tokens));
165  // parse 'else'
166  block.push_back(parse(tokens));
167  }
168  else if (token.token == "let")
169  {
170  auto temp = tokens.front();
171  // parse identifier
172  if (temp.type == TokenType::Identifier)
173  block.push_back(atom(nextToken(tokens)));
174  else
175  throwParseError("need an identifier to define a constant", temp);
176  // value
177  block.push_back(parse(tokens));
178  }
179  else if (token.token == "mut")
180  {
181  auto temp = tokens.front();
182  // parse identifier
183  if (temp.type == TokenType::Identifier)
184  block.push_back(atom(nextToken(tokens)));
185  else
186  throwParseError("need an identifier to define a variable", temp);
187  // value
188  block.push_back(parse(tokens));
189  }
190  else if (token.token == "set")
191  {
192  auto temp = tokens.front();
193  // parse identifier
194  if (temp.type == TokenType::Identifier)
195  block.push_back(atom(nextToken(tokens)));
196  else
197  throwParseError("need an identifier to set the value of a variable", temp);
198  // value
199  block.push_back(parse(tokens));
200  }
201  else if (token.token == "fun")
202  {
203  // parse arguments
204  if (tokens.front().type == TokenType::Grouping)
205  {
206  block.push_back(parse(tokens, /* authorize_capture */ true));
207  }
208  else
209  throwParseError("invalid token to define an arguments list", tokens.front());
210  // parse body
211  if (tokens.front().type == TokenType::Grouping)
212  block.push_back(parse(tokens));
213  else
214  throwParseError("invalid token to define the body of a function", tokens.front());
215  }
216  else if (token.token == "while")
217  {
218  auto temp = tokens.front();
219  // parse condition
220  if (temp.type == TokenType::Grouping)
221  block.push_back(parse(tokens));
222  else if (temp.type == TokenType::Identifier || temp.type == TokenType::Number ||
223  temp.type == TokenType::String)
224  block.push_back(atom(nextToken(tokens)));
225  else
226  throwParseError("ill-formed while, couldn't find condition", temp);
227  // parse 'do'
228  block.push_back(parse(tokens));
229  }
230  else if (token.token == "begin")
231  {
232  while (true)
233  {
234  except(!tokens.empty(), "No more token to consume when creating begin block", m_last_token);
235  if (tokens.front().token == ")")
236  break;
237  m_last_token = tokens.front();
238 
239  block.push_back(parse(tokens));
240  }
241  }
242  else if (token.token == "import")
243  {
244  if (tokens.front().type == TokenType::String)
245  block.push_back(atom(nextToken(tokens)));
246  else
247  throwParseError("invalid import rule: should be a string", tokens.front());
248  }
249  else if (token.token == "quote")
250  {
251  block.push_back(parse(tokens));
252  }
253  else if (token.token == "del")
254  {
255  if (tokens.front().type == TokenType::Identifier)
256  block.push_back(atom(nextToken(tokens)));
257  else
258  throwParseError("invalid token: del can only be applied to identifers", tokens.front());
259  }
260  }
261  else if (token.type == TokenType::Identifier || token.type == TokenType::Operator ||
262  (token.type == TokenType::Capture && authorize_capture) ||
263  (token.type == TokenType::GetField && authorize_field_read))
264  {
265  while (tokens.front().token != ")")
266  block.push_back(parse(tokens, /* authorize_capture */ false, /* authorize_field_read */ true));
267  }
268  } while (tokens.front().token != ")");
269 
270  // pop the ")"
271  tokens.pop_front();
272  return block;
273  }
274  else if (token.type == TokenType::Shorthand)
275  {
276  if (token.token == "'")
277  {
278  // create a list node to host the block
279  Node block(NodeType::List);
280  block.setPos(token.line, token.col);
281 
282  block.push_back(Node(Keyword::Quote));
283  block.push_back(parse(tokens));
284  return block;
285  }
286  else
287  throwParseError("unknown shorthand", token);
288  }
289  return atom(token);
290  }

302  Node Parser::atom(const Token& token)
303  {
304  if (token.type == TokenType::Number)
305  {
306  auto n = Node(std::stod(token.token));
307  n.setPos(token.line, token.col);
308  return n;
309  }
310  else if (token.type == TokenType::String)
311  {
312  std::string str = token.token;
313  // remove the " at the beginning and at the end
314  str.erase(0, 1);
315  str.erase(token.token.size() - 2, 1);
316 
317  Ark::Utils::stringReplaceAll(str, "\\t", "\t");
318  Ark::Utils::stringReplaceAll(str, "\\n", "\n");
319  Ark::Utils::stringReplaceAll(str, "\\r", "\r");
320  Ark::Utils::stringReplaceAll(str, "\\v", "\v");
321 
322  auto n = Node(str);
323  n.setPos(token.line, token.col);
324  return n;
325  }
326  else if (token.type == TokenType::Keyword)
327  {
328  std::optional<Keyword> kw;
329  if (token.token == "if") kw = Keyword::If;
330  else if (token.token == "set") kw = Keyword::Set;
331  else if (token.token == "let") kw = Keyword::Let;
332  else if (token.token == "mut") kw = Keyword::Mut;
333  else if (token.token == "fun") kw = Keyword::Fun;
334  else if (token.token == "while") kw = Keyword::While;
335  else if (token.token == "begin") kw = Keyword::Begin;
336  else if (token.token == "import") kw = Keyword::Import;
337  else if (token.token == "quote") kw = Keyword::Quote;
338  else if (token.token == "del") kw = Keyword::Del;
339  if (kw)
340  {
341  auto n = Node(kw.value());
342  n.setPos(token.line, token.col);
343  return n;
344  }
345  throwParseError("unknown keyword", token);
346  }
347  else if (token.type == TokenType::Capture)
348  {
349  auto n = Node(NodeType::Capture);
350  n.setString(token.token);
351  n.setPos(token.line, token.col);
352  return n;
353  }
354  else if (token.type == TokenType::GetField)
355  {
356  auto n = Node(NodeType::GetField);
357  n.setString(token.token);
358  n.setPos(token.line, token.col);
359  return n;
360  }
361 
362  // assuming it is a TokenType::Identifier, thus a Symbol
363  auto n = Node(NodeType::Symbol);
364  n.setString(token.token);
365  n.setPos(token.line, token.col);
366  return n;
367  }

370  bool Parser::checkForInclude(Node& n)
371  {
372  if (n.nodeType() == NodeType::Keyword)
373  {
374  if (n.keyword() == Keyword::Import)
375  return true;
376  }
377  else if (n.nodeType() == NodeType::List)
378  {
379  // can not optimize calls to n.list().size() because we are modifying n.list()
380  for (std::size_t i=0; i < n.list().size(); ++i)
381  {
382  if (checkForInclude(n.list()[i]))
383  {
384  if (m_debug >= 2)
385  Ark::logger.info("Import found in file:", m_file);
386 
387  std::string file;
388  if (n.const_list()[1].nodeType() == NodeType::String)
389  file = n.const_list()[1].string();
390  else
391  throw Ark::TypeError("Arguments of import must be of type String");
392 
393  namespace fs = std::filesystem;
394  using namespace std::string_literals;
395 
396  std::string ext = fs::path(file).extension().string();
397  std::string dir = Ark::Utils::getDirectoryFromPath(m_file) + "/";
398  std::string path = (dir != "/") ? dir + file : file;
399 
400  if (m_debug >= 2)
401  Ark::logger.data("path:", path, " ; file:", file, " ; libdir:", m_libdir);
402 
403  // check if we are not loading a plugin
404  if (ext == ".ark")
405  {
406  n.list().clear();
407  // replace content with a begin block
408  n.list().emplace_back(Keyword::Begin);
409 
410  // lib paths
411  std::string libpath = m_libdir + "/" + Ark::Utils::getFilenameFromPath(file);
412  std::string libpath2 = m_libdir + "/" + file;
413 
414  std::string included_file = "";
415  // search in the files of the user first
416  if (Ark::Utils::fileExists(path))
417  included_file = path;
418  else if (Ark::Utils::fileExists(libpath))
419  included_file = libpath;
420  else if (Ark::Utils::fileExists(libpath2))
421  included_file = libpath2;
422  else
423  throw std::runtime_error("ParseError: Couldn't find file " + file);
424 
425  // if the file isn't in the include list, then we can include it
426  // this avoids cyclic includes
427  if (std::find(m_parent_include.begin(), m_parent_include.end(), included_file) == m_parent_include.end())
428  {
429  Parser p(m_debug, m_libdir, m_options);
430  // feed the new parser with our parent includes
431  for (auto&& pi : m_parent_include)
432  p.m_parent_include.push_back(pi); // new parser, we can assume that the parent include list is empty
433  p.m_parent_include.push_back(m_file); // add the current file to avoid importing it again
434  p.m_parent_include.push_back(included_file);
435 
436  p.feed(Ark::Utils::readFile(included_file), included_file);
437 
438  // update our list of included files
439  for (auto&& inc : p.m_parent_include)
440  {
441  if (std::find(m_parent_include.begin(), m_parent_include.end(), inc) == m_parent_include.end())
442  m_parent_include.push_back(inc);
443  }
444 
445  n.list().push_back(p.ast());
446  }
447  else if (m_debug >= 1)
448  Ark::logger.warn("Possible cyclic inclusion issue: file " + m_file + " is trying to include " + path + " which was already included");
449  }
450  }
451  }
452  }
453 
454  return false;
455  }
