420  else if (token.token == "let") kw = Keyword::Let;
421  else if (token.token == "mut") kw = Keyword::Mut;
422  else if (token.token == "fun") kw = Keyword::Fun;
423  else if (token.token == "while") kw = Keyword::While;
424  else if (token.token == "begin") kw = Keyword::Begin;
425  else if (token.token == "import") kw = Keyword::Import;
426  else if (token.token == "quote") kw = Keyword::Quote;
427  else if (token.token == "del") kw = Keyword::Del;
428 
429  if (kw)
430  return make_node(kw.value(), token.line, token.col, m_file);
431  throwParseError("unknown keyword", token);
432  }
433 
434  case TokenType::Capture:
435  case TokenType::GetField:
436  case TokenType::Spread:
437  {
438  Node n = make_node(similar_nodetype_from_tokentype(token.type), token.line, token.col, m_file);

440  return n;

478

481  {

485  // if the file isn't in the include list, then we can include it
486  // this avoids cyclic includes

491  // feed the new parser with our parent includes
492  for (auto&& pi : m_parent_include)
493  p.m_parent_include.push_back(pi); // new parser, we can assume that the parent include list is empty
494  p.m_parent_include.push_back(m_file); // add the current file to avoid importing it again
495  p.feed(Ark::Utils::readFile(included_file), included_file);
496 
497  // update our list of included files
498  for (auto&& inc : p.m_parent_include)
499  {
500  if (std::find(m_parent_include.begin(), m_parent_include.end(), inc) == m_parent_include.end())
501  m_parent_include.push_back(inc);
502  }
503 
504  for (std::size_t j = 1, end = p.ast().constList().size(); j < end; ++j)
505  {
506  parent.list().insert(parent.list().begin() + pos + j, p.ast().constList()[j]);
507  }
508 
509  return true;
510  }
511  }
512 
513  for (std::size_t i = 0; i < n.list().size(); ++i)
514  {
515  if (checkForInclude(n.list()[i], n, i))
516  {
517  n.list().erase(n.list().begin() + i);
518  --i;
519  }
520  }
521  }
522
